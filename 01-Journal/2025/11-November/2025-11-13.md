```dataviewjs
const currentFile = dv.current().file.name;
const currentPath = dv.current().file.path;
const dateMatch = currentFile.match(/\d{4}-\d{2}-\d{2}/);

if (!dateMatch) {
  dv.paragraph("⚠️ File name not in format YYYY-MM-DD.");
} else {
  const date = dateMatch[0]; 
  const normalizedDate = date.replace(/-/g, "");

  const pages = dv.pages()
    .where(p => {
      if (!p.created) return false;
      const createdStr = String(p.created)
        .replace(/\D/g, "")  
        .substring(0, 8);
      if (p.file.path === currentPath) return false;
      return createdStr === normalizedDate;
    })
    .sort(p => p.file.name, 'asc');

  dv.header(3, `Other notes`);

  if (pages.length === 0) {
    dv.paragraph(`No file other notes.`);
  } else {
    dv.table(
      ["Title", "Type", "Tags"],
      pages.map(p => {
          let label = "";
    if (typeof p.title === "string" && p.title.trim() !== "") {
        label = p.title.trim();
    } else {
        label = p.file.name;
    }
      
     return  [
        dv.fileLink(p.file.path, false, label),
        p.type ?? "",
        (p.tags && p.tags.length > 0) ? p.tags.join(", ") : ""
      ]})
    );
  }
}
```
